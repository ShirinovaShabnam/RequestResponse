CREATE TABLE albums (
  id INT PRIMARY KEY AUTO_INCREMENT,
  album VARCHAR(100) NOT NULL,
  `year`INT,
	band_id INT NOT NULL,
	FOREIGN KEY (band_id) REFERENCES bands(id)
);
